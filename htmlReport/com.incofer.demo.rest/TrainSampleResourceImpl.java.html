package com.incofer.demo.rest;

import com.incofer.demo.entity.TrainEntity;
import com.incofer.demo.model.Train;
import com.incofer.demo.services.TrainService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.beans.factory.annotation.Qualifier;
import org.springframework.http.HttpStatus;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.*;

import java.util.List;

@RestController
public class TrainSampleResourceImpl {


    @Autowired
    @Qualifier("trainService")
    private TrainService trainService;

    public void setTrainService(TrainService trainService) {
        this.trainService = trainService;
    }

    public TrainService getTrainService() {
        return trainService;
    }

    @PostMapping("/save")
    public Train saveTrain(@RequestBody Train train) throws Exception {
        return this.trainService.save(train);
    }

    @GetMapping("/search")
    public List<TrainEntity> searchTrain(@RequestParam String name) throws Exception {
        return this.trainService.findByName(name);
    }

    @GetMapping("/hello")
    public String sayHello() {
        return "Hello, world!";
    }

    @GetMapping("/listRoutes")
    public String listRoutes() {
        return "list routes";
    }

    @GetMapping("/Train")
    public Train save(Train train) {
        return train;
    }

    @DeleteMapping("/delete")
    public ResponseEntity<String> deleteByName(@RequestParam String name) throws Exception {
        this.trainService.deleteByName(name);
        return new ResponseEntity<>("Train deleted successfully", HttpStatus.OK);

    }

    @PutMapping("/update")
    public ResponseEntity<String> updateTrain(@RequestBody Train train) {
        try {
           this.trainService.update(train);
            return new ResponseEntity<>("Train updated successfully", HttpStatus.OK);
        } catch (Exception e) {
            return new ResponseEntity<>(e.getMessage(), HttpStatus.BAD_REQUEST);
        }


    }
}
