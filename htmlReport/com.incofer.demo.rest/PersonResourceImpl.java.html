package com.incofer.demo.rest;
import com.incofer.demo.persistence.PersonPersistence;
import com.incofer.demo.persistence.impl.PersonJpaPersistenceImpl;
import com.incofer.demo.model.Person;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.*;

@RestController
@RequestMapping("/persons")
public class PersonResourceImpl
{
    private final PersonPersistence personPersistence;

    @Autowired
    public PersonResourceImpl(PersonPersistence personDao)
    {
        this.personPersistence = personDao;
    }

    @PostMapping
    public Person createPerson(@RequestBody Person person)
    {
        return personPersistence.save(person);
    }

    @GetMapping("/{id}")
    public Person getPersonById(@PathVariable int id) throws PersonJpaPersistenceImpl.PersonDaoException
    {
        return personPersistence.findById(id);
    }

    @PutMapping("/{id}")
    public void updatePerson(@PathVariable int id, @RequestBody Person updatedPerson) throws PersonJpaPersistenceImpl.PersonDaoException
    {
        Person person = personPersistence.findById(id);
        person.setName(updatedPerson.getName());
        personPersistence.update(person);
    }

    @DeleteMapping("/{id}")
    public void deletePerson(@PathVariable int id) throws PersonJpaPersistenceImpl.PersonDaoException {
        Person person = personPersistence.findById(id);
        personPersistence.delete(person);
    }
}
