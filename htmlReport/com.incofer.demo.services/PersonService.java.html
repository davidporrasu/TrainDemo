package com.incofer.demo.services;

import com.incofer.demo.persistence.PersonPersistence;
import com.incofer.demo.persistence.impl.PersonJpaPersistenceImpl;
import com.incofer.demo.model.Person;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

@Service
public class PersonService
{
    private final PersonPersistence personPersistence;

    @Autowired
    public PersonService(final PersonPersistence personPersistence)
    {
        this.personPersistence = personPersistence;
    }


    public Person save(final Person person)
    {
        return personPersistence.save(person);
    }

    public Person findById(final int id) throws PersonJpaPersistenceImpl.PersonDaoException
    {
        return personPersistence.findById(id);
    }

    public void update(final Person person)
    {
        personPersistence.update(person);
    }

    public void delete(final Person person) throws PersonJpaPersistenceImpl.PersonDaoException
    {
        personPersistence.delete(person);
    }
}
