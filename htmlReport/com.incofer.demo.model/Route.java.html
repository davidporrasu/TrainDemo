package com.incofer.demo.model;

import com.incofer.demo.enums.City;
import lombok.AllArgsConstructor;
import lombok.Builder;
import lombok.Data;
import lombok.NoArgsConstructor;
import lombok.extern.slf4j.Slf4j;

@Data
@NoArgsConstructor
@AllArgsConstructor
@Slf4j
@Builder

public class Route
{
    private City origin = City.SAN_JOSE;
    private City station1 = City.CURRIDABAT;
    private City station2 = City.TRES_RIOS;
    private City station3 = City.OCHOMOGO;
    private City finalDestination = City.CARTAGO;
    private int kilometersTraveled = 28;
    private int boardingTime;
    private double stationTime = 0.78;
    private double durationTime = 37.5;
    private int trainVelocity= 50;
    private int id = 1;
    public final static int TIME_PER_KM = 2;


    public double getTimeFrom1To2()
    {
       double timeToStation1 = ( City.SAN_JOSE.getKm() + City.CURRIDABAT.getKm() ) * Route.TIME_PER_KM;
       return timeToStation1;
    }

    public double getTimeFrom2To3()
    {
        double timeToStation2 = ( City.TRES_RIOS.getKm() - City.CURRIDABAT.getKm() )  * Route.TIME_PER_KM;
        return timeToStation2;
    }
    public double getTimeFrom3To4()
    {
        double timeToStation3 = (City.OCHOMOGO.getKm() - City.TRES_RIOS.getKm()) * Route.TIME_PER_KM;
        return timeToStation3;
    }
    public double getTimeFrom4To5()
    {
        double timeToStation4 = (City.CARTAGO.getKm() - City.OCHOMOGO.getKm()) * Route.TIME_PER_KM;
        return timeToStation4;
    }
    public double getRouteTotalTime()
    {
        double timeToStation1 = getTimeFrom1To2();
        double timeToStation2 = getTimeFrom2To3();
        double timeToStation3 = getTimeFrom3To4();
        double timeToStation4 = getTimeFrom4To5();
        return timeToStation1 + timeToStation2 + timeToStation3 + timeToStation4;
    }
}
