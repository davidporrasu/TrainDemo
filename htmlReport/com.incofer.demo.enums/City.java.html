package com.incofer.demo.enums;

@SuppressWarnings("checkstyle:RegexpMultiline")
public enum City {

    CARTAGO("Cartago", 28),
    SAN_JOSE("San Jose", 0),
    CURRIDABAT("uaca", 8),
    TRES_RIOS("Tres Rios", 12),
    OCHOMOGO("Ochomogo", 22);

    private int km;
    private String displayText;

    City ( final String pText, final int km )
    {
        this.displayText = pText;
        this.km = km;
    }
    public int getKm()
    {
        return this.km;
    }
}
