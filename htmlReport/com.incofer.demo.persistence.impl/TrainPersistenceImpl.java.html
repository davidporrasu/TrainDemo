package com.incofer.demo.persistence.impl;

import com.incofer.demo.persistence.TrainPersistence;

import com.incofer.demo.model.Train;
import org.springframework.stereotype.Repository;
import org.springframework.web.context.annotation.ApplicationScope;

import java.util.ArrayList;
import java.util.List;

@Repository("trainPersistenceImpl")
@ApplicationScope
public class TrainPersistenceImpl implements TrainPersistence
{
    private List<Train> trains = new ArrayList<>();

    @Override
    public Train save(Train train)
    {
        trains.add(train);
        return train;
    }

    @Override
    public Train search(String train)
    {
        int index = this.trains.indexOf(train);
        return (Train) this.trains.get(index);
    }

    @Override
    public void deleteByName(String name)
    {
        trains.removeIf(train -> train.getName().equals(name));
    }

    @Override
    public void update(Train train)
    {
        for (int i = 0; i < this.trains.size(); i++)
        {
            if (this.trains.get(i).getName().equals(train.getName()))
            {
                this.trains.set(i, train);
                break;
            }
        }

    }
    @Override
    public List<Train> findByName(final String name)
    {
        List<Train> arraylist = new ArrayList<>();
        for (Train train : trains)
        {
            if (train.getName().equals(name))
            {
                arraylist.add(train);
            }
        }
        return arraylist;
    }

}
