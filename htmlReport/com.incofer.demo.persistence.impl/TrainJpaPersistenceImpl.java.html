package com.incofer.demo.persistence.impl;

import com.incofer.demo.entity.TrainEntity;
import com.incofer.demo.model.Train;
import com.incofer.demo.persistence.TrainPersistence;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.stereotype.Repository;
import org.springframework.transaction.annotation.Transactional;

import javax.persistence.EntityManager;
import javax.persistence.PersistenceContext;
import java.util.ArrayList;
import java.util.List;

@Repository("trainJpaPersistenceImpl")
@Transactional
public class TrainJpaPersistenceImpl implements TrainPersistence
{
    private static final Logger logger = LoggerFactory.getLogger(TrainJpaPersistenceImpl.class);
    @PersistenceContext
    private EntityManager entityManager;

    @SuppressWarnings("checkstyle:FinalLocalVariable")
    @Override
    public Train save(final Train train)
    {
        TrainEntity entity = new TrainEntity();
        entity.setName(train.getTrainName());
        entityManager.persist(entity);
        logger.info("Train saved successfully, Train Details=" + train);
        return train;
    }

    @Override
    public Train search(final String name)
    {
        TrainEntity trainEntity = entityManager.find(TrainEntity.class, name);
        if (trainEntity == null)
        {
            //throw new TrainDaoException("Train with id " + name + " not found", name);
            logger.info("Train NOT found, Train Details=" + trainEntity);
        }
        logger.info("Train found successfully, Train Details=" + trainEntity);

        Train train = new Train();
        train.setTrainName(trainEntity.getName());

        return train;
    }

    @Override
    public void update(final Train train)
    {

    }

    @Override
    public List<Train> findByName(final String name)
    {
        List<Train> results = new ArrayList<>();
        TrainEntity trainEntity = entityManager.find(TrainEntity.class, name);
        if (trainEntity == null)
        {
            //throw new TrainDaoException("Train with id " + name + " not found", name);
            logger.info("Train NOT found, Train Details=" + trainEntity);
        }
        else
        {
            logger.info("Train found successfully, Train Details=" + trainEntity);
            Train train = new Train();
            train.setTrainName(trainEntity.getName());
            train.setTrainCabin(99);
            results.add(train);
        }

        return results;
    }
