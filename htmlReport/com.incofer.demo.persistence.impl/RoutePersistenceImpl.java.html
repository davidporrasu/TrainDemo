package com.incofer.demo.persistence.impl;

import com.incofer.demo.model.Route;
import com.incofer.demo.enums.City;

import java.util.ArrayList;
import java.util.Collection;
import java.util.List;
import java.util.Optional;

public class RoutePersistenceImpl
{
    private final List<Route> routes = new ArrayList<>();

    public RoutePersistenceImpl()
    {
        // add some default routes
        routes.add(new Route());
        routes.add(new Route());
    }

    public void save(final Route route) throws IllegalArgumentException
    {
        if (route == null)
        {
            throw new IllegalArgumentException("Route is null");
        }
        routes.add(route);
    }

    public void delete(final Route route) throws IllegalArgumentException
    {
        if (route == null)
        {
            throw new IllegalArgumentException("Route is null");
        }
        if (!routes.contains(route))
        {
            throw new IllegalArgumentException("Route does not exist");
        }
        routes.remove(route);
    }

    public Optional<Route> search(final City origin, final City finalDestination)
    {
        for (Route route : routes)
        {
            if (route.getOrigin() == origin && route.getFinalDestination() == finalDestination)
            {
                // Calcular los tiempos de cada tramo
                double timeFrom1To2 = route.getTimeFrom1To2();
                double timeFrom2To3 = route.getTimeFrom2To3();
                double timeFrom3To4 = route.getTimeFrom3To4();
                double timeFrom4To5 = route.getTimeFrom4To5();

                double totalTime = timeFrom1To2 + timeFrom2To3 + timeFrom3To4 + timeFrom4To5;

                // Establecer el tiempo total en la ruta
                route.setDurationTime(totalTime);

                return Optional.of(route);
            }
        }
        return Optional.empty();
    }

    public void update(final Route route, final City origin, final City station1, final City station2, final City station3, final City finalDestination) throws IllegalArgumentException
    {
        if (route == null)
        {
            throw new IllegalArgumentException("Route is null");
        }
        if (!routes.contains(route))
        {
            throw new IllegalArgumentException("Route does not exist");
        }
        route.setOrigin(origin);
        route.setStation1(station1);
        route.setStation2(station2);
        route.setStation3(station3);
        route.setFinalDestination(finalDestination);
    }

    public Route getRouteById(final int id) throws IllegalArgumentException
    {
        for (Route route : routes)
        {
            if (route.getId() == id)
            {
                return route;
            }
        }
        throw new IllegalArgumentException("Route with id " + id + " does not exist");
    }

    public void setRouteById(final int id, final Route updatedRoute) throws IllegalArgumentException
    {
        for (int i = 0; i < routes.size(); i++)
        {
            if (routes.get(i).getId() == id)
            {
                routes.set(i, updatedRoute);
                return;
            }
        }
        throw new IllegalArgumentException("Route with id " + id + " does not exist");
    }
    public Collection<Object> getAllRoutes()
    {
        return null;
    }
}
